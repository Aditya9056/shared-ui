Labels styles can get applied using sui-label class but it's better to use them with form fields.

<!-- Default label -->
<label class="sui-label">Label</label>

<!-- Label with link (at right) -->
<label class="sui-label">
	Label
	<a href="#" class="sui-label-link">Side link</a>
</label>

<!-- Label with note (at right) -->
<label class="sui-label">
	Label
	<span class="sui-label-note">Side note</span>
</label>
